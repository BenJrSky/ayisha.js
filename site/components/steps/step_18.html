4.2 Nested Conditional Logic

<select @model="userRole">
  <option value="">Select Role</option>
  <option value="guest">Guest</option>
  <option value="user">User</option>
  <option value="admin">Admin</option>
</select>

<div @if="userRole">
  <div @if="userRole === 'guest'">
    <p>Welcome, Guest! Limited access.</p>
    <div @if="guestActions">
      <button @click="guestActions = false">Disable Guest Actions</button>
    </div>
    <div @if="!guestActions">
      <button @click="guestActions = true">Enable Guest Actions</button>
    </div>
  </div>
  
  <div @if="userRole === 'user'">
    <p>Welcome, User! You have standard access.</p>
    <div @show="showUserPanel">
      <h4>User Panel</h4>
      <p>Your profile settings and preferences.</p>
    </div>
    <button @click="showUserPanel = !showUserPanel">
      {{showUserPanel ? 'Hide' : 'Show'}} User Panel
    </button>
  </div>
  
  <div @if="userRole === 'admin'">
    <p>Welcome, Admin! Full system access.</p>
    <div @if="adminMode === 'basic'">
      <p>Basic admin tools loaded.</p>
      <button @click="adminMode = 'advanced'">Switch to Advanced Mode</button>
    </div>
    <div @if="adminMode === 'advanced'">
      <p style="color:red;">⚠️ Advanced admin tools - use with caution!</p>
      <button @click="adminMode = 'basic'">Switch to Basic Mode</button>
    </div>
  </div>
</div>

<div @if="!userRole">
  <p style="color:#666;">Please select a role to see available options.</p>
</div>

🎯 What you learned:
  • @if completely removes/adds elements from DOM
  • @show/@hide just toggle CSS display property
  • Nested @if directives create complex conditional logic
  • Use @show for frequent toggles, @if for heavy content
