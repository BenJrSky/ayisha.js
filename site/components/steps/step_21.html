7.3 Real-World Example: Dynamic Search

<!-- Search with debounced input -->
<input @model="searchTerm" 
       @input="searchTerm.length > 2 ? (searchUsers = true) : (searchUsers = false)"
       placeholder="Search users (type at least 3 characters)">
       
<div @if="searchUsers && searchTerm.length > 2"
     @fetch="'https://jsonplaceholder.typicode.com/users'"
     @result="allUsers"></div>
     
<div @if="allUsers"
     @source="allUsers"
     @filter="user => user.name.toLowerCase().includes(searchTerm.toLowerCase()) || user.email.toLowerCase().includes(searchTerm.toLowerCase())"
     @result="filteredUsers"></div>
