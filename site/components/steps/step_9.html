9.2. Navigazione Breadcrumb

<!-- Breadcrumb navigation -->
<nav class="breadcrumb">
  <span @for="crumb in breadcrumbs">
    <a @click="navigateTo(crumb.path)">{{crumb.name}}</a>
    <span @if="index < breadcrumbs.length - 1"> / </span>
  </span>
