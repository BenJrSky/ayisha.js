todos = JSON.parse(localStorage.getItem('ayisha-todos') || '[]'); newTodo = ''; filter = 'all'; saveTodos = function() { localStorage.setItem('ayisha-todos', JSON.stringify(todos)); }; addTodo = function() { if (newTodo.trim()) { todos.push({ id: Date.now(), text: newTodo.trim(), completed: false, createdAt: new Date().toLocaleDateString() }); newTodo = ''; saveTodos(); } }; toggleTodo = function(id) { const todo = todos.find(t => t.id === id); if (todo) { todo.completed = !todo.completed; saveTodos(); } }; removeTodo = function(id) { todos = todos.filter(t => t.id !== id); saveTodos(); }; getFilteredTodos = function() { if (filter === 'active') return todos.filter(t => !t.completed); if (filter === 'completed') return todos.filter(t => t.completed); return todos; }; clearCompleted = function() { todos = todos.filter(t => !t.completed); saveTodos(); };
