showBasicModal = false; showConfirmDialog = false; showFormModal = false; showImageModal = false; showNotification = false; selectedImage = ''; formData = { name: '', email: '', message: '' }; notification = { type: '', message: '' }; pendingAction = null; // Funzioni per aprire i vari modali openBasicModal = function() { showBasicModal = true; }; openConfirmDialog = function(action) { pendingAction = action; showConfirmDialog = true; }; openFormModal = function() { formData = { name: '', email: '', message: '' }; showFormModal = true; }; openImageModal = function(imageSrc) { selectedImage = imageSrc; showImageModal = true; }; // Funzioni per chiudere i modali closeModal = function() { showBasicModal = false; showConfirmDialog = false; showFormModal = false; showImageModal = false; pendingAction = null; }; // Gestione conferma confirmAction = function() { if (pendingAction) { showNotification = true; notification = { type: 'success', message: `Azione "${pendingAction}" eseguita con successo!` }; setTimeout(() => { showNotification = false; }, 3000); } closeModal(); }; // Gestione form submitForm = function() { if (formData.name && formData.email && formData.message) { showNotification = true; notification = { type: 'success', message: `Messaggio inviato da ${formData.name}!` }; setTimeout(() => { showNotification = false; }, 3000); closeModal(); } else { showNotification = true; notification = { type: 'error', message: 'Compila tutti i campi richiesti!' }; setTimeout(() => { showNotification = false; }, 3000); } }; // Gestione click overlay handleOverlayClick = function(event) { if (event.target === event.currentTarget) { closeModal(); } };
