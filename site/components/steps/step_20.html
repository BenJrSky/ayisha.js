6.4. Chain Operations for Complex Data Processing

<!-- Multi-step data pipeline -->
<!-- Step 1: Filter products by category -->
<div @source="products" @filter="p => p.category === 'Electronics'" @result="electronics"></div>
<!-- Step 2: Add calculated fields -->
<div @source="electronics" @map="p => ({...p, totalValue: p.price * p.quantity, discount: p.price > 500 ? 0.1 : 0})" @result="processedElectronics"></div>
<!-- Step 3: Apply discounts -->
<div @source="processedElectronics" @map="p => ({...p, finalPrice: p.price * (1 - p.discount)})" @result="finalElectronics"></div>
<!-- Step 4: Calculate total savings -->
<div @source="finalElectronics" @reduce="(acc, p) => acc + (p.price * p.discount * p.quantity)" @initial="0" @result="totalSavings"></div>
