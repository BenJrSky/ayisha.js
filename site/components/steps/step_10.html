10.1. Fetch API di Base

<!-- Fetch e gestione dati -->
<script>
async function fetchData() {
  loading = true;
  try {
    const response = await fetch('https://api.example.com/data');
    const data = await response.json();
    items = data;
  } catch (error) {
    errorMessage = 'Errore nel caricamento';
  } finally {
    loading = false;
  }
}
</script>

<button @click="fetchData()">Carica Dati</button>
<div @if="loading">Caricamento...</div>
<div @if="errorMessage">{{errorMessage}}</div>
