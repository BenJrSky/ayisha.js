11.2. Validazione Multi-Step

<!-- Validazione complessa multi-campo -->
<script>
function validateStep(stepNumber) {
  const validators = {
    1: () => validatePersonalInfo(),
    2: () => validateAddress(),
    3: () => validatePayment()
  };
  
  const result = validators[stepNumber]?.() || { valid: false };
  
  if (!result.valid) {
    showErrors(result.errors);
    return false;
  }
  
  return true;
}
</script>
