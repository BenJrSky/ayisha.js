    function addPost() {
      const post = { userId: state.userId, title: state.newTitle, body: state.newBody };
      fetch('https://jsonplaceholder.typicode.com/posts', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(post)
      })
      .then(res => res.json())
      .then(data => {
        state.lastPost = data;
        state.posts = [data, ...(state.posts||[])];
        state.newTitle = '';
        state.newBody = '';
      });
